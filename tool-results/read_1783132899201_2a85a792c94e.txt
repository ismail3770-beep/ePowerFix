     1→// ePowerFix SQLite schema (converted from PostgreSQL).
     2→// - Provider: sqlite (no enums, no @db.*, no Decimal, no Json)
     3→// - Enums -> String (UPPERCASE literal values kept)
     4→// - Decimal -> Float
     5→// - Json -> String (apps must JSON.parse/stringify manually)
     6→// - String[] (scalar lists) are supported by the SQLite connector since Prisma 4.
     7→
     8→generator client {
     9→  provider = "prisma-client-js"
    10→}
    11→
    12→datasource db {
    13→  provider = "sqlite"
    14→  url      = env("DATABASE_URL")
    15→}
